73 void gaussWithFlow(cv::InputArray input, cv::OutputArray output, const cv::Mat& vfield,
74  int ksize, double sigma_s, double sigma_t) {
75  cv::Mat image = input.getMat();
76  cv::Mat& out = output.getMatRef();
77 
78  const int width = image.cols;
79  const int height = image.rows;
80  const int dim = image.channels();
81  const int L = static_cast<int>(ksize * 1.5);
82  out = cv::Mat::zeros(height, width, CV_MAKETYPE(CV_32F, dim));

84  cv::Mat temp = cv::Mat::zeros(height, width, CV_MAKETYPE(CV_32F, dim));
85 
86  ompfor(int y = 0; y < height; y++) {
87  for (int x = 0; x < width; x++) {
88  double tx = vfield.at<float>(y, x * 2 + 0);
89  double ty = vfield.at<float>(y, x * 2 + 1);
90  double weight = 0.0;
91  for (int t = -ksize; t <= ksize; t++) {
92  int xx = static_cast<int>(x - 0.5 * ty * t);
93  int yy = static_cast<int>(y + 0.5 * tx * t);
94  if (xx >= 0 && yy >= 0 && xx < width && yy < height) {
95  double w = exp(-t * t / sigma_t);
96  for (int c = 0; c < dim; c++) {
97  temp.at<float>(y, x*dim + c) += static_cast<float>(w * image.at<float>(yy, xx*dim + c));
98  }
99  weight += w;
100  }
101  }
102 
103  for (int c = 0; c < dim; c++) {
104  temp.at<float>(y, x*dim + c) = temp.at<float>(y, x*dim + c) / static_cast<float>(weight);
105  }
106  }
107  }

109  ompfor(int y = 0; y < height; y++) {
110  for (int x = 0; x < width; x++) {
111  double weight = 0.0;
112  for (int c = 0; c < dim; c++) {
113  out.at<float>(y, x*dim + c) = 0.0f;
114  }
115 
116  for (int pm = -1; pm <= 1; pm += 2) {
117  int l = 0;
118  double tx = pm * vfield.at<float>(y, x * 2 + 0);
119  double ty = pm * vfield.at<float>(y, x * 2 + 1);
120  Point2d pt = Point2d(x + 0.5, y + 0.5);
121  while (++l < L) {
122  int px = static_cast<int>(ceil(pt.x));
123  int py = static_cast<int>(ceil(pt.y));
124  if (px < 0 || py < 0 || px >= width || py >= height) {
125  break;
126  }
127 
128  double w = exp(-l * l / sigma_s);
129  for (int c = 0; c < dim; c++) {
130  out.at<float>(y, x*dim + c) += static_cast<float>(w * temp.at<float>(py, px*dim + c));
131  }
132  weight += w;
133 
134  double vx = vfield.at<float>(py, px * 2 + 0);
135  double vy = vfield.at<float>(py, px * 2 + 1);
136  if (vx == 0.0f && vy == 0.0f) {
137  break;
138  }
139 
140  double inner = vx * tx + vy * ty;
141  double sx = sign(inner) * vx;
142  double sy = sign(inner) * vy;
143  px = static_cast<int>(ceil(pt.x + 0.5 * sx));
144  py = static_cast<int>(ceil(pt.y + 0.5 * sy));
145  if (px < 0 || py < 0 || px >= width || py >= height) {
146  break;
147  }
148 
149  vx = vfield.at<float>(py, px * 2 + 0);
150  vy = vfield.at<float>(py, px * 2 + 1);
151  inner = vx * tx + vy * ty;
152  tx = sign(inner) * vx;
153  ty = sign(inner) * vy;
154  pt.x += tx;
155  pt.y += ty;
156  }
157  }
158 
159  for (int c = 0; c < dim; c++) {
160  if (weight != 0.0) {
161  out.at<float>(y, x*dim + c) = out.at<float>(y, x*dim + c) / static_cast<float>(weight);
162  } else {
163  out.at<float>(y, x*dim + c) = temp.at<float>(y, x*dim + c);
164  }
165  }
166  }
167  }
