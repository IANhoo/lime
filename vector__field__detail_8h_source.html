35 void calcETF(cv::InputArray input, cv::OutputArray output,
36  int ksize = 5, int maxiter = 3, EdgeDetector edgeDetector = EDGE_SOBEL) {
37  cv::Mat image = input.getMat();
38  cv::Mat& etf = output.getMatRef();
39 
40  // check input arguments
41  int width = image.cols;
42  int height = image.rows;
43  int dim = image.channels();
44  static const double eps = 1.0e-8;
45 
46  cv::Mat gray;
47  if (image.depth() != CV_32F) {
48  image.convertTo(gray, CV_MAKETYPE(CV_32F, dim), 1.0 / 255.0);
49  } else {
50  image.convertTo(gray, CV_MAKETYPE(CV_32F, dim));
51  }
52 
53  if (dim != 1) {
54  cv::cvtColor(gray, gray, cv::COLOR_BGR2GRAY);
55  }
56 
57  // detect standard edge
58  cv::Mat gx, gy;
59  if (edgeDetector == EDGE_SOBEL) {
60  cv::Sobel(gray, gx, CV_32FC1, 1, 0);
61  cv::Sobel(gray, gy, CV_32FC1, 0, 1);
62  } else if (edgeDetector == EDGE_ROTATIONAL) {
63  double p1 = 0.183;
64  double Dx[3][3] = { { 0.5 * p1, 0.0, -0.5 * p1 },
65  { 0.5 - p1, 0.0, p1 - 0.5 },
66  { 0.5 * p1, 0.0, -0.5 * p1 } };
67 
68  gx = cv::Mat(height, width, CV_32FC1);
69  gy = cv::Mat(height, width, CV_32FC1);
70  for (int y = 0; y < height; y++) {
71  for (int x = 0; x < width; x++) {
72  double sumx = 0.0;
73  double sumy = 0.0;
74  double weightx = 0.0;
75  double weighty = 0.0;
76  for (int dy = -1; dy <= 1; dy++) {
77  for (int dx = -1; dx <= 1; dx++) {
78  int xx = x + dx;
79  int yy = y + dy;
80  if (xx >= 0 && yy >= 0 && xx < width && yy < height) {
81  double wx = Dx[dy + 1][dx + 1];
82  double wy = Dx[dx + 1][dy + 1];
83  sumx += wx * gray.at<float>(yy, xx);
84  sumy += wy * gray.at<float>(yy, xx);
85  }
86  }
87  }
88  gx.at<float>(y, x) = static_cast<float>(sumx);
89  gy.at<float>(y, x) = static_cast<float>(sumy);
90  }
91  }
92  } else {
93  Assertion(false, "Unknown edge detector is specified");
94  }
95 
96  // compute tangent field
97  cv::Mat tangent = cv::Mat::zeros(height, width, CV_32FC2);
98  cv::Mat ghat = cv::Mat::zeros(height, width, CV_32FC1);
99  for (int y = 0; y < height; y++) {
100  for (int x = 0; x < width; x++) {
101  double vx = gx.at<float>(y, x);
102  double vy = gy.at<float>(y, x);
103  double mag = sqrt(vx*vx + vy*vy) + eps;
104  ghat.at<float>(y, x) = static_cast<float>(mag);
105 
106  tangent.at<float>(y, x * 2 + 0) = static_cast<float>(-vy / mag);
107  tangent.at<float>(y, x * 2 + 1) = static_cast<float>(vx / mag);
108  }
109  }
110 
111  // compute ETF
112  cv::Mat temp = cv::Mat(height, width, CV_32FC2);
113  while (maxiter--) {
114  ompfor(int y = 0; y < height; y++) {
115  for (int x = 0; x < width; x++) {
116  Point2d sum = Point2d(0.0, 0.0);
117  double weight = 0.0;
118  for (int dy = -ksize; dy <= ksize; dy++) {
119  for (int dx = -ksize; dx <= ksize; dx++) {
120  int xx = x + dx;
121  int yy = y + dy;
122  if (xx < 0 || yy < 0 || xx >= width || yy >= height) continue;
123  if (dx * dx + dy * dy > ksize * ksize) continue;
124 
125  double vx = tangent.at<float>(y, x * 2 + 0);
126  double vy = tangent.at<float>(y, x * 2 + 1);
127  Point2d tx = Point2d(vx, vy);
128 
129  double ux = tangent.at<float>(yy, xx * 2 + 0);
130  double uy = tangent.at<float>(yy, xx * 2 + 1);
131  Point2d ty = Point2d(ux, uy);
132  double wd = tx.dot(ty);
133 
134  double gx = ghat.at<float>(y, x);
135  double gy = ghat.at<float>(yy, xx);
136  double wm = (gy - gx + 1.0) / 2.0;
137 
138  sum = sum + ty * (wm * wd);
139  weight += wm * wd;
140  }
141  }
142 
143  Point2d v = weight != 0.0 ? sum / weight : Point2d(0.0, 0.0);
144  temp.at<float>(y, x * 2 + 0) = static_cast<float>(v.x);
145  temp.at<float>(y, x * 2 + 1) = static_cast<float>(v.y);
146  }
147  }
148 
149  temp.convertTo(tangent, CV_32FC2);
150  for (int y = 0; y < height; y++) {
151  for (int x = 0; x < width; x++) {
152  double vx = tangent.at<float>(y, x * 2 + 0);
153  double vy = tangent.at<float>(y, x * 2 + 1);
154  double mag = sqrt(vx * vx + vy * vy) + eps;
155  tangent.at<float>(y, x * 2 + 0) = static_cast<float>(vx / mag);
156  tangent.at<float>(y, x * 2 + 1) = static_cast<float>(vy / mag);
157  }
158  }
159  }
160 
161  // compute tangent direction with angle
162  temp.convertTo(etf, CV_32F);
163 }

165 void calcSST(cv::InputArray input, cv::OutputArray output, int ksize = 5, EdgeDetector edgeDetector = EDGE_SOBEL) {
166  cv::Mat image = input.getMat();
167  cv::Mat& sst = output.getMatRef();
168 
169  int width = image.cols;
170  int height = image.rows;
171  int dim = image.channels();
172 
173  const int offset[4][2] = { { 1, 0 }, { 0, 1 }, { -1, 0 }, { 0, -1 } };
174 
175  cv::Mat gray;
176  if (image.depth() != CV_32F) {
177  image.convertTo(gray, CV_MAKETYPE(CV_32F, dim), 1.0 / 255.0);
178  } else {
179  image.convertTo(gray, CV_MAKETYPE(CV_32F, dim));
180  }
181 
182  if (dim != 1) {
183  cv::cvtColor(gray, gray, cv::COLOR_BGR2GRAY);
184  }
185 
186  // detect standard edge
187  cv::Mat gx, gy;
188  if (edgeDetector == EDGE_SOBEL) {
189  cv::Sobel(gray, gx, CV_32FC1, 1, 0);
190  cv::Sobel(gray, gy, CV_32FC1, 0, 1);
191  } else if (edgeDetector == EDGE_ROTATIONAL) {
192  double p1 = 0.183;
193  double Dx[3][3] = { { 0.5 * p1, 0.0, -0.5 * p1 },
194  { 0.5 - p1, 0.0, p1 - 0.5 },
195  { 0.5 * p1, 0.0, -0.5 * p1 } };
196 
197  gx = cv::Mat(height, width, CV_32FC1);
198  gy = cv::Mat(height, width, CV_32FC1);
199  for (int y = 0; y < height; y++) {
200  for (int x = 0; x < width; x++) {
201  double sumx = 0.0;
202  double sumy = 0.0;
203  double weightx = 0.0;
204  double weighty = 0.0;
205  for (int dy = -1; dy <= 1; dy++) {
206  for (int dx = -1; dx <= 1; dx++) {
207  int xx = x + dx;
208  int yy = y + dy;
209  if (xx >= 0 && yy >= 0 && xx < width && yy < height) {
210  double wx = Dx[dy + 1][dx + 1];
211  double wy = Dx[dx + 1][dy + 1];
212  sumx += wx * gray.at<float>(yy, xx);
213  sumy += wy * gray.at<float>(yy, xx);
214  }
215  }
216  }
217  gx.at<float>(y, x) = static_cast<float>(sumx);
218  gy.at<float>(y, x) = static_cast<float>(sumy);
219  }
220  }
221  } else {
222  Assertion(false, "Unknown edge detector is specified.");
223  }
224 
225  // tensor relaxation step
226  cv::Mat efg = cv::Mat(height, width, CV_64FC3);
227  for (int y = 0; y < height; y++) {
228  for (int x = 0; x < width; x++) {
229  double fx = gx.at<float>(y, x);
230  double fy = gy.at<float>(y, x);
231  efg.at<double>(y, x * 3 + 0) = fx * fx;
232  efg.at<double>(y, x * 3 + 1) = fx * fy;
233  efg.at<double>(y, x * 3 + 2) = fy * fy;
234  }
235  }
236 
237  cv::Mat tmp;
238  efg.convertTo(tmp, CV_64FC3);
239  const double tau = 0.002;
240  for (int it = 0; it < 5; it++) {
241  for (int y = 0; y < height; y++) {
242  for (int x = 0; x < width; x++) {
243  double E = efg.at<double>(y, x * 3 + 0);
244  double F = efg.at<double>(y, x * 3 + 1);
245  double G = efg.at<double>(y, x * 3 + 2);
246 
247  double eng = sqrt(E * E + 2.0 * F * F + G * G);
248  if (eng > tau) {
249  double sumE = 0.0;
250  double sumF = 0.0;
251  double sumG = 0.0;
252  double cnt = 0.0;
253  for (int k = 0; k < 4; k++) {
254  int xx = x + offset[k][0];
255  int yy = y + offset[k][1];
256  if (xx >= 0 && yy >= 0 && xx < width && yy < height) {
257  sumE += tmp.at<double>(yy, xx * 3 + 0);
258  sumF += tmp.at<double>(yy, xx * 3 + 1);
259  sumG += tmp.at<double>(yy, xx * 3 + 2);
260  cnt += 1.0;
261  }
262  }
263  efg.at<double>(y, x * 3 + 0) = sumE / cnt;
264  efg.at<double>(y, x * 3 + 1) = sumF / cnt;
265  efg.at<double>(y, x * 3 + 2) = sumG / cnt;
266  }
267  }
268  }
269  efg.convertTo(tmp, CV_64FC3);
270  }
271 
272  efg.convertTo(efg, CV_32F);
273  double sigma = 2.0 * ksize * ksize;
274  cv::GaussianBlur(efg, sst, cv::Size(ksize, ksize), sigma);
275 }

279 void calcVectorField(cv::InputArray input, cv::OutputArray angles,
280  int ksize, VFieldType vfieldType, EdgeDetector edgeDetector) {
281  cv::Mat image = input.getMat();
282 
283  const int width = image.cols;
284  const int height = image.rows;
285 
286  cv::Mat& vfield = angles.getMatRef();
287  vfield = cv::Mat(height, width, CV_32FC1);
288  if (vfieldType == VECTOR_SST) {
289  cv::Mat sst;
290  npr::calcSST(image, sst, ksize, edgeDetector);
291 
292  for (int y = 0; y < height; y++) {
293  for (int x = 0; x < width; x++) {
294  double E = sst.at<float>(y, x * 3 + 0);
295  double F = sst.at<float>(y, x * 3 + 1);
296  double G = sst.at<float>(y, x * 3 + 2);
297  double D = sqrt((E - G) * (E - G) + 4.0 * F * F);
298  double lambda1 = (E + G + D) / 2.0;
299  double lambda2 = (E + G - D) / 2.0;
300  double lambda = std::abs(lambda1) > std::abs(lambda2) ? lambda1 : lambda2;
301  vfield.at<float>(y, x) = static_cast<float>(atan2(-F, lambda1 - E));
302  }
303  }
304  } else if (vfieldType == VECTOR_ETF) {
305  cv::Mat etf;
306  npr::calcETF(image, etf, ksize, edgeDetector);
307  for (int y = 0; y < height; y++) {
308  for (int x = 0; x < width; x++) {
309  double tx = etf.at<double>(y, x * 2 + 0);
310  double ty = etf.at<double>(y, x * 2 + 1);
311  vfield.at<float>(y, x) = static_cast<float>(atan2(ty, tx));
312  }
313  }
314  } else {
315  Assertion(false, "Unknown vector field type is specified.");
316  }
317 }

319 void detectSingular(const Array2d<Tensor>& sst, std::vector<SingularPoint>* points,
320  std::vector<cv::Point2f>* delauneyNodes) {
321  const int width = sst.cols();
322  const int height = sst.rows();
323 
324  if (delauneyNodes->empty()) {
325  cv::Mat white = cv::Mat::ones(height, width, CV_32FC1);
326  lime::npr::poissonDisk(white, delauneyNodes, lime::npr::PDS_FAST_PARALLEL, 3.0, 5.0);
327  }
328 
329  cv::Rect(0, 0, width, height);
330  cv::Subdiv2D subdiv(cv::Rect(0, 0, width, height));
331  for (int i = 0; i < delauneyNodes->size(); i++) {
332  subdiv.insert(delauneyNodes[i]);
333  }
334 
335  std::vector<cv::Vec6f> tri;
336  subdiv.getTriangleList(tri);
337  for (int i = 0; i < tri.size(); i++) {
338  cv::Mat A = cv::Mat::ones(3, 3, CV_64FC1);
339  int x1 = static_cast<int>(tri[i][0]);
340  int y1 = static_cast<int>(tri[i][1]);
341  if (x1 < 0 || y1 < 0 || x1 >= width || y1 >= height) continue;
342  A.at<double>(0, 0) = sst(y1, x1).elem(0, 0) - sst(y1, x1).elem(1, 1);
343  A.at<double>(1, 0) = sst(y1, x1).elem(0, 1);
344 
345  int x2 = static_cast<int>(tri[i][2]);
346  int y2 = static_cast<int>(tri[i][3]);
347  if (x2 < 0 || y2 < 0 || x2 >= width || y2 >= height) continue;
348  A.at<double>(0, 1) = sst(y2, x2).elem(0, 0) - sst(y2, x2).elem(1, 1);
349  A.at<double>(1, 1) = sst(y2, x2).elem(0, 1);
350 
351  int x3 = static_cast<int>(tri[i][4]);
352  int y3 = static_cast<int>(tri[i][5]);
353  if (x3 < 0 || y3 < 0 || x3 >= width || y3 >= height) continue;
354  A.at<double>(0, 2) = sst(y3, x3).elem(0, 0) - sst(y3, x3).elem(1, 1);
355  A.at<double>(1, 2) = sst(y3, x3).elem(0, 1);
356 
357  cv::Mat b = cv::Mat::zeros(3, 1, CV_64FC1);
358  b.at<double>(2, 0) = 1.0;
359 
360  cv::Mat p = cv::Mat(b.size(), CV_64FC1);
361  if (cv::determinant(A) != 0) {
362  cv::solve(A, b, p);
363  double p1 = p.at<double>(0, 0);
364  double p2 = p.at<double>(1, 0);
365  double p3 = p.at<double>(2, 0);
366  if (p1 >= 0.0 && p2 >= 0.0 && p3 >= 0.0) {
367  int px = static_cast<int>(p1 * x1 + p2 * x2 + p3 * x3);
368  int py = static_cast<int>(p1 * y1 + p2 * y2 + p3 * y3);
369  if (px >= 1 && py >= 1 && px < width - 1 && py < height - 1) {
370  // determine type of singularity
371  Tensor Tx = (sst(py, px + 1) - sst(py, px - 1)) * 0.5;
372  Tensor Ty = (sst(py + 1, px) - sst(py - 1, px)) * 0.5;
373  double h11 = 0.25 * (Tx.elem(0, 0) - Tx.elem(1, 1));
374  double h21 = 0.5 * Tx.elem(0, 1);
375  double h12 = 0.25 * (Ty.elem(0, 0) - Ty.elem(1, 1));
376  double h22 = 0.5 * Ty.elem(0, 1);
377  double delta = h11 * h22 - h12 * h21;
378  double idx = 1 - 0.5 * (2.0 - sign(delta));
379  if (sign(delta) > 0) {
380  points->push_back(SingularPoint(px, py, 0.0, SINGULAR_WEDGE));
381  } else if (sign(delta) < 0) {
382  points->push_back(SingularPoint(px, py, 0.0, SINGULAR_TRISECTOR));
383  } else {
384  points->push_back(SingularPoint(px, py, 0.0, SINGULAR_HIGHER_ORDER));
385  }
386  }
387  }
388  }
389  }
